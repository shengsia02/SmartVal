def gen_buyers(n=1300):
    rows = []
    for _ in range(n):
        name = chinese_name()
        rows.append({
            "姓名": name,
            "聯絡電話": taiwan_mobile(),
            "電子郵件": f"{romanize(name)}@mail.com".lower()
        })
    return pd.DataFrame(rows)

def gen_houses(n, agents_df, buyers_df):
    cities = list(city_districts.keys())
    city_dist_pairs = [(c, d) for c in cities for d in city_districts[c]]
    random.shuffle(city_dist_pairs)
    rows = []
    for i in range(n):
        city, dist = city_dist_pairs[i % len(city_dist_pairs)]
        address = random_address(city, dist)
        htype = random.choice(house_types)
        total_price = price_by_city(city)
        floor_area = round(random.uniform(12, 90), 2)
        land_area = round(max(5.0, floor_area * random.uniform(0.3, 1.5)), 2)
        unit_price = round(total_price / max(floor_area, 10), 2)
        age = random.randint(0, 50)
        start = datetime(2019,1,1)
        sold_date = (start + timedelta(days=random.randint(0, (datetime(2025,11,4)-start).days))).date().isoformat()
        agent_name = agents_df.sample(1, ignore_index=True).iloc[0]["姓名"]
        buyer_name = buyers_df.sample(1, ignore_index=True).iloc[0]["姓名"]
        rows.append({
            "地址": address,
            "房屋類型": htype,
            "總價格（萬）": total_price,
            "縣市": city,
            "行政區": dist,
            "屋齡": age,
            "建坪": floor_area,
            "地坪": land_area,
            "單價（萬/坪）": unit_price,
            "出售日期": sold_date,
            "仲介": agent_name,
            "買家": buyer_name
        })
    return pd.DataFrame(rows)
